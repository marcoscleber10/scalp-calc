// ===== FUNÇÃO DO ROBÔ - ORDEM REAL (SEM MENSAGEM DE ESPERA) ===== async function placeRealOrder() { if(!hlConnection){ alert('Conecte a API primeiro'); return; } const capital = parseFloat(document.getElementById('botCapital').value); const pair = document.getElementById('botPair').value; const sl = parseFloat(document.getElementById('botSL').value)/100; const tp = parseFloat(document.getElementById('botTP').value)/100; if(capital < 5){ alert('Capital mínimo: 5 USDC'); return; } if(!price || price === 0){ alert('Aguardando preço...'); return; } try { // Mapeamento dos 6 pares const assetMap = { BTC: 0, ETH: 1, SOL: 2, GOLD: 3, SILVER: 4, WTI: 5 }; const assetId = assetMap[pair]; if (assetId === undefined) { alert('Par inválido'); return; } const size = (capital / price).toFixed(6); const priceStr = price.toFixed(2); const nonce = Date.now() + Math.floor(Math.random() * 1000); const action = { type: "order", orders: [{ a: assetId, b: true, p: priceStr, r: false, s: size, t: { limit: { tif: "Gtc" } } }], grouping: "na" }; const domain = { name: "Hyperliquid", version: "1", chainId: 42161 }; const types = { Action: [ { name: "type", type: "string" }, { name: "orders", type: "Order[]" }, { name: "grouping", type: "string" } ], Order: [ { name: "a", type: "uint32" }, { name: "b", type: "bool" }, { name: "p", type: "string" }, { name: "r", type: "bool" }, { name: "s", type: "string" }, { name: "t", type: "OrderType" } ], OrderType: [ { name: "limit", type: "Limit" } ], Limit: [ { name: "tif", type: "string" } ] }; document.getElementById('apiConsole').innerHTML = '⏳ Assinando ordem...'; const signature = await hlConnection.apiWallet._signTypedData( domain, types, action ); const sig = ethers.utils.splitSignature(signature); const requestBody = { action, nonce, signature: { r: sig.r, s: sig.s, v: sig.v }, vaultAddress: null }; document.getElementById('apiConsole').innerHTML = '⏳ Enviando ordem...'; const response = await fetch('https://api.hyperliquid.xyz/exchange', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(requestBody) }); const result = await response.json(); if(result.status === "ok" || result.response?.data?.statuses?.[0]?.resting){ document.getElementById('apiConsole').innerHTML = '✅ Ordem enviada com sucesso!'; const stopPrice = price * (1 - sl); const takePrice = price * (1 + tp); alert(`✅ ORDEM REAL EXECUTADA!\n\n` + `Comprar ${size} ${pair}\n` + `Preço: $${price.toFixed(2)}\n` + `Stop: $${stopPrice.toFixed(2)}\n` + `Take: $${takePrice.toFixed(2)}`); } else { document.getElementById('apiConsole').innerHTML = '❌ ' + JSON.stringify(result); } } catch(e){ document.getElementById('apiConsole').innerHTML = '❌ ' + e.message; } }
