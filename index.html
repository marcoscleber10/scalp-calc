// ===== FUNÇÃO DO ROBÔ CORRIGIDA ===== async function placeRealOrder() { if(!hlConnection){ alert('Conecte a API primeiro'); return; } const capital = parseFloat(document.getElementById('botCapital').value); const pair = document.getElementById('botPair').value; const sl = parseFloat(document.getElementById('botSL').value)/100; const tp = parseFloat(document.getElementById('botTP').value)/100; if(capital < 5){ alert('Capital mínimo: 5 USDC'); return; } if(!price || price === 0){ alert('Aguardando preço...'); return; } try { const assetMap = { BTC:0, ETH:1, SOL:2, GOLD:3, SILVER:4, WTI:5 }; const assetId = assetMap[pair]; if (assetId === undefined) { alert('Par inválido'); return; } // 🔥 Direção baseada no sinal do filtro técnico const signal = document.getElementById('rsiText').innerText; const isBuy = signal.includes('COMPRA'); const size = (capital / price).toFixed(6); const priceStr = price.toFixed(2); const nonce = Date.now(); const action = { type: "order", orders: [{ a: assetId, b: isBuy, p: priceStr, s: size, r: false, t: { limit: { tif: "Gtc" } } }], grouping: "na" }; const domain = { name: "Exchange", version: "1", chainId: 42161 }; const types = { Order: [ { name: "a", type: "uint32" }, { name: "b", type: "bool" }, { name: "p", type: "string" }, { name: "s", type: "string" }, { name: "r", type: "bool" }, { name: "t", type: "OrderType" } ], OrderType: [ { name: "limit", type: "Limit" } ], Limit: [ { name: "tif", type: "string" } ], Action: [ { name: "type", type: "string" }, { name: "orders", type: "Order[]" }, { name: "grouping", type: "string" } ] }; document.getElementById('apiConsole').innerHTML = '⏳ Assinando...'; const signature = await hlConnection.apiWallet._signTypedData(domain, types, action); const sig = ethers.utils.splitSignature(signature); const requestBody = { action, nonce, signature: { r: sig.r, s: sig.s, v: sig.v }, vaultAddress: null }; document.getElementById('apiConsole').innerHTML = '⏳ Enviando...'; const response = await fetch('https://api.hyperliquid.xyz/exchange', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(requestBody) }); const result = await response.json(); if(result.status === "ok" || result.response?.data?.statuses?.[0]?.resting){ const stopPrice = isBuy ? price * (1 - sl) : price * (1 + sl); const takePrice = isBuy ? price * (1 + tp) : price * (1 - tp); document.getElementById('apiConsole').innerHTML = '✅ Ordem enviada!'; alert(`🚀 ORDEM EXECUTADA Tipo: ${isBuy ? 'COMPRA' : 'VENDA'} Ativo: ${pair} Quantidade: ${size} Preço: $${price.toFixed(2)} TP: $${takePrice.toFixed(2)} SL: $${stopPrice.toFixed(2)}`); } else { document.getElementById('apiConsole').innerHTML = '❌ Erro: ' + JSON.stringify(result); } } catch(e){ document.getElementById('apiConsole').innerHTML = '❌ ' + e.message; } }
